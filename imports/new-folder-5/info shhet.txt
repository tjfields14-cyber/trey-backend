# pip install openai==1.* beautifulsoup4 requests readability-lxml
import os, json, time, requests
from bs4 import BeautifulSoup
from readability import Document
from openai import OpenAI

OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")  # set this in your environment
client = OpenAI(api_key=OPENAI_API_KEY)

def fetch_url(url, timeout=20):
    r = requests.get(url, headers={"User-Agent":"Mozilla/5.0 (compatible; TJF-miner/1.0)"}, timeout=timeout)
    r.raise_for_status()
    return r.text

def extract_main_text(html):
    doc = Document(html)
    cleaned = doc.summary(html_partial=True)
    soup = BeautifulSoup(cleaned, "html.parser")
    text = soup.get_text("\n", strip=True)
    title = doc.short_title()
    return title, text

def ai_extract(title, text):
    # Structured JSON fields you want from each page
    system = "You extract structured data from web pages. Respond ONLY valid JSON."
    user = f"""Title: {title}

Content:
{text[:12000]}  # truncate long pages

Return JSON with:
- title
- topics: list of 3-6 tags
- summary: 3-5 bullet points
- entities: list of {{type, name}} (people/orgs/places)
- sentiment: one of ["positive","neutral","negative"]
"""
    resp = client.responses.create(
        model="gpt-4.1-mini",
        input=[{"role":"system","content":system},{"role":"user","content":user}],
        temperature=0.2
    )
    # Responses API returns a top-level 'output_text' helper on SDK;
    # fallback: read the first text item from the 'output' array.
    out = resp.output_text  # see docs if your SDK version differs
    return json.loads(out)

def mine(urls):
    results = []
    for u in urls:
        try:
            html = fetch_url(u)
            title, text = extract_main_text(html)
            data = ai_extract(title, text)
            data["source_url"] = u
            results.append(data)
            time.sleep(1)  # be polite
        except Exception as e:
            results.append({"source_url": u, "error": str(e)})
    return results

if __name__ == "__main__":
    seeds = [
        "https://example.com/",
        # add more URLs (or load from a file)
    ]
    mined = mine(seeds)
    with open("miner_output.json", "w", encoding="utf-8") as f:
        json.dump(mined, f, ensure_ascii=False, indent=2)
    print("Saved miner_output.json")
